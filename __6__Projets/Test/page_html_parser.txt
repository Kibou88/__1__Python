			function createXmlHttpRequest() {
				var req = null;
				if (window.XMLHttpRequest) {
					req = new XMLHttpRequest();
					if (req.overrideMimeType) {
						req.overrideMimeType('text/xml');
					}
				} else if (window.ActiveXObject) {
					try {
						req = new ActiveXObject("Msxml2.XMLHTTP");
					} catch (e)	{

							req = new ActiveXObject("Microsoft.XMLHTTP");
						} catch (e) {
							return null;
						}
					}
				}
				return req;
			}

			function submitForm() {
				var submit_req = null;
				var userinfo_req = null;
				var params = [];
				var paramline = "";
				var url = "";
				var forminputs = [];
				var formselects = [];
				var count = 0;
				var i;

				var max_userinfo_req = 300;
				var period_userinfo_req = 3000;

				try {
					// Try to create XHR object depending of OS and Browser
					//  * For form submit
					setWarning("Sending...");
					submit_req = createXmlHttpRequest();
					if (submit_req === null) {
						setError("ERROR");
						return false;
					}
					submit_req.open("POST", "sponsor.html", true);
					// * For userinfo.html request
					userinfo_req = createXmlHttpRequest();
					if (userinfo_req === null) {
						setError("ERROR");
						return false;
					}
					userinfo_req.open("GET", "userinfo.html", true);

					// Form submit callback
					submit_req.onreadystatechange = function() {
						// readyState === DONE
						if (submit_req.readyState === 4)	{
							// http status code === OK
							if (submit_req.status === 200) {
								setWarning("Pending validation of the request");
								userinfo_req.send(null);
							} else if(submit_req.status === 429) {
								// http status code === TOO MANY REQUESTS
								setError("ERROR: Brute force attack detected");
							} else {
								activate_submit();
								setError("ERROR");
							}
						}
					}

					// Userinfo request callback
					userinfo_req.onreadystatechange = function() {
						var nb_tries;

						// readyState === DONE
						if (userinfo_req.readyState === 4) {
							nb_tries = count++;
							// http status code === OK
							if (userinfo_req.status === 200) {
								window.setTimeout("window.location=\"/auth/userinfo.html\"", 5000);
								setWarning("Your request was validated. " + "<a class=\"redirectlink\" href=\"/auth/userinfo.html\">&raquo; Click here if you are not automatically redirected</a>");
							} else if (userinfo_req.status === 401) {
								// http status code === UNAUTHORIZED
								if ( limit_userinfo_req && (nb_tries >= max_userinfo_req) ) {
									// no more try (timeout)
									activate_submit();
									setError("Your request was not validated");
									$('#info').css('display', 'block');
									return false;
								}
								window.setTimeout(function() { userinfo_req.send(null); }, period_userinfo_req);
								userinfo_req.open("GET", "userinfo.html", true);
							} else {
								activate_submit();
								setError("ERROR");
							}
						}
					};

					// Construct form submit POST request

					forminputs = $('input');
					for (i = 0; i < forminputs.length; i++) {

							paramline += "&";
						paramline += $(forminputs[i]).attr('name') + "=" + escape($(forminputs[i]).val());
					}

					// * Add select tags to params
					formselects = $('select');
					for (i = 0; i < formselects.length; i++) {
						if (paramline.length !== 0)
							paramline += "&";
						paramline += $(formselects[i]).attr('name') + "=" + escape($(formselects[i]).val());
					}

					// Send POST form
					deactivate_submit()
					submit_req.send(paramline);

					// Cancel browser submit event
					return false;
				} catch (e) {
					setError("ERROR");
					return false;
				}
